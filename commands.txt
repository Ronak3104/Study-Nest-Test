# intialisation of backend
npm init -y

# backend dependencies
npm install express mongoose dotenv cors helmet morgan bcryptjs jsonwebtoken cloudinary multer stripe axios
npm install --save-dev nodemon
npm install joi express-rate-limit

# intialisation of frontend
npm create vite@latest . -- --template react
npm install

# frontend dependencies
npm install lucide-react recharts @stripe/react-stripe-js @stripe/stripe-js axios zustand react-router-dom
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p
npm install react-hot-toast @tailwindcss/postcss

# Python dependencies
python -m venv venv
source venv/bin/activate     # On Windows: venv\Scripts\activate
pip install -r requirements.txt

uvicorn app.generator:app --host 0.0.0.0 --port 8000 --reload
